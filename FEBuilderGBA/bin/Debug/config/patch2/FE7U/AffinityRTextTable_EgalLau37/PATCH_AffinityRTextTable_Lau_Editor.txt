IF:$FGREP4 AffinityTableCode.bin=0x05 0x48 0xC0 0x68

//STRUCTパッチとして実装します
TYPE=STRUCT
TAG=#ENGINE

//データへのポインタ
POINTER=$FGREP4END AffinityTableCode.bin

//データサイズ(10進数)
DATASIZE=2

//データ個数 00 が出てくるまで
DATACOUNT=0x8

//リストに表示されるのは属性アイコンです
LISTNAME=ATTRIBUTEICON0

//データを定義します
W0:TEXT=Text for affinity
