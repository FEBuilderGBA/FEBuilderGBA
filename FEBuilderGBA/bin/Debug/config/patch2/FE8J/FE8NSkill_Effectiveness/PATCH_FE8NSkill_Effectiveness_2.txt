NAME=FE8NSkill 魔物特効(SkillID:0x1C 魔物特効)の対象クラスの指定

IF:$FGREP4 main.dmp=0x00 0xB5 0x30 0xB4

//STRUCTパッチとして実装します
TYPE=STRUCT
TAG=#ENGINE

//データへのポインタ
POINTER=$FGREP4END+20 main.dmp

//データサイズ(10進数)

//データ個数 00 が出てくるまで
DATACOUNT=$GREP1 0x00

//リストに表示されるのはクラス名です
LISTNAME=$B0:CLASS

//データを定義します
B0:CLASS=CLASS
